cmake_minimum_required(VERSION 3.12.0)

# setup
project(Engine LANGUAGES CXX C)
add_executable(${PROJECT_NAME} main.cpp)
set(EXECUTABLE_OUTPUT_PATH ${CMAKE_SOURCE_DIR}/bin)
set_property(TARGET ${PROJECT_NAME} PROPERTY CXX_STANDARD 20)

# additional dirs
include_directories(vendor/include src)
add_subdirectory(src/Core)
add_subdirectory(src/OpenGL)
add_subdirectory(src/ImGui)

# glfw3
add_library(GLFW3 STATIC IMPORTED GLOBAL)
set_property(TARGET GLFW3 PROPERTY IMPORTED_LOCATION "${CMAKE_CURRENT_SOURCE_DIR}/vendor/lib/GLFW/glfw3.lib")
target_link_libraries(${PROJECT_NAME} GLFW3)

# glad
target_sources(${PROJECT_NAME} PUBLIC vendor/src/glad/glad.c)

# stb_image
target_sources(${PROJECT_NAME} PUBLIC vendor/src/stb_image/stb_image.cpp)
